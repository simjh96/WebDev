middleware(Express에서만 사용)
	req, res cycle에 끼어 들수 있는 함수
		next를 써줘야 cycle이 계속됨

 	reqTime(req, res, next) { req.조작; next(); };
	app.use(reqTime);
	app.get(()=>{여기 안에서 자유롭게 사용 가능})

	app.use(express.static())   # 이런것들도 다 middleware
